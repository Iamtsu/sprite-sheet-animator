3. Export your animation definitions as JSON
4. Load the JSON file here to see your animations in action!

Example Setup

This demo creates animations programmatically. You can do the same in your projects: 

const animator = new SpriteSheetAnimator('sprite.png', {
    frameWidth: 64,
    frameHeight: 64
});

await animator.waitForLoad();

animator.addAnimationFromGrid('walk', {
    row: 0,
    startCol: 0,
    frameCount: 8,
    frameRate: 12,
    loop: true
});

animator.play('walk');

// In game loop
function gameLoop(timestamp) {
    animator.update(deltaTime);
    animator.draw(ctx, x, y);
    requestAnimationFrame(gameLoop);
}
